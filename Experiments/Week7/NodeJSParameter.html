var express = require('express') var app = express() app.get('/paramSend/:param', function (req, res) { var paramInput = req.params.param; res.send(paramInput); }); var ip = process.env.OPENSHIFT_NODEJS_IP || '127.0.0.1'; var port = process.env.OPENSHIFT_NODEJS_PORT || 3000; app.listen(port, ip);
